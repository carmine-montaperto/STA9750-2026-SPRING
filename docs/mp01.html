acquire_ipeds_data <- function(start_year = 2010, end_year = 2024){
  library(tidyverse)
  library(glue)

  data_dir <- file.path("data", "mp01")

  if(!dir.exists(data_dir)){
    dir.create(data_dir, showWarnings = FALSE, recursive = TRUE)
  }

  YEARS <- seq(start_year, end_year)

  EFA_ALL <- purrr::map(YEARS, function(yy){
    if(yy <= 2022){
      ef_url <- glue("https://nces.ed.gov/ipeds/datacenter/data/EF{yy}A.zip")
    } else {
      ef_url <- glue("https://nces.ed.gov/ipeds/data-generator?year={yy}&tableName=EF{yy}A&HasRV=0&type=csv")
    }

    ef_file <- file.path(data_dir, glue("ef{yy}a.csv.zip"))

    if(!file.exists(ef_file)){
      message(glue("Downloading Enrollment Data for {yy} from {ef_url}"))
      download.file(ef_url, destfile = ef_file, quiet = TRUE, mode = "wb")
    }

    readr::read_csv(
      ef_file,
      show_col_types = FALSE
    ) |>
      dplyr::mutate(
        year = yy,
        enrollment_m_aian = EFAIANM,
        enrollment_f_aian = EFAIANW,
        enrollment_m_asia = EFASIAM,
        enrollment_f_asia = EFASIAW,
        enrollment_m_bkaa = EFBKAAM,
        enrollment_f_bkaa = EFBKAAW,
        enrollment_m_hisp = EFHISPM,
        enrollment_f_hisp = EFHISPW,
        enrollment_m_nhpi = EFNHPIM,
        enrollment_f_nhpi = EFNHPIW,
        enrollment_m_whit = EFWHITM,
        enrollment_f_whit = EFWHITW,
        enrollment_m_2mor = EF2MORM,
        enrollment_f_2mor = EF2MORW,
        enrollment_m_unkn = EFUNKNM,
        enrollment_f_unkn = EFUNKNW,
        enrollment_m_nral = EFNRALM,
        enrollment_f_nral = EFNRALW
      ) |>
      dplyr::filter(
        (EFALEVEL %in% c(2, 12)) | (LINE %in% c(1, 15))
      ) |>
      dplyr::mutate(
        level = dplyr::case_when(
          EFALEVEL == 2 ~ "all undergrad",
          EFALEVEL == 12 ~ "all graduate",
          LINE %in% c(1, 15) ~ "first year undergrad"
        )
      ) |>
      dplyr::select(
        institution_id = UNITID,
        year,
        level,
        dplyr::starts_with("enrollment_")
      ) |>
      dplyr::group_by(institution_id, year, level) |>
      dplyr::summarize(
        dplyr::across(dplyr::starts_with("enrollment_"), sum),
        .groups = "drop"
      )
  }) |>
    dplyr::bind_rows()

  DESC_ALL <- purrr::map(YEARS, function(yy){
    if(yy <= 2022){
      hd_url <- glue("https://nces.ed.gov/ipeds/datacenter/data/HD{yy}.zip")
    } else {
      hd_url <- glue("https://nces.ed.gov/ipeds/data-generator?year={yy}&tableName=HD{yy}&HasRV=0&type=csv")
    }

    hd_file <- file.path(data_dir, glue("hd{yy}.csv.zip"))

    if(!file.exists(hd_file)){
      message(glue("Downloading Institutional Descriptions for {yy} from {hd_url}"))
      download.file(hd_url, destfile = hd_file, quiet = TRUE, mode = "wb")
    }

    suppressWarnings(
      readr::read_csv(
        hd_file,
        show_col_types = FALSE,
        locale = readr::locale(
          encoding = dplyr::if_else(yy == 2024, "utf-8", "windows-1252")
        )
      ) |>
        dplyr::mutate(
          year = yy,
          INSTNM = INSTNM
        ) |>
        dplyr::select(
          institution_id = UNITID,
          institution_name = INSTNM,
          state = STABBR,
          year
        )
    )
  }) |>
    dplyr::bind_rows()

  dplyr::inner_join(
    EFA_ALL,
    DESC_ALL,
    dplyr::join_by(institution_id == institution_id, year == year)
  )
}

Op-Ed

The Supreme Court’s 2023 ruling in Students for Fair Admissions v. Harvard ended the use of race-conscious admissions nationwide, raising understandable questions about how universities will maintain diverse learning environments going forward. For institutions like Rutgers, which serve a broad public mission, the decision creates an opportunity to reflect on how we define diversity and how we measure progress over time. While the legal framework around admissions has changed, the broader goal of building inclusive campuses remains an important part of higher education.

To understand how these national changes might intersect with Rutgers’ own trajectory, I examined IPEDS enrollment data for Rutgers–New Brunswick from 2010 through 2024. The dataset includes both all undergraduate students and first-year undergraduates, which makes it possible to compare long-term institutional trends with the more immediate signals that appear in incoming classes. Because IPEDS is publicly reported federal data, it provides a consistent way to track changes in student composition over time.

Looking at the full 2010–2024 period, Rutgers–New Brunswick shows a steady increase in undergraduate diversity. The diversity share for all undergraduates rises from about 25.7% in 2010 to 36.5% in 2024, representing a gradual upward trend over more than a decade. Rather than sudden shifts, the data suggests incremental growth as the university’s student body evolved over time.

The most noticeable change appears in the newest first-year class. First-year diversity increases from 34.5% in 2023 to 38.7% in 2024, a larger jump than the overall undergraduate population’s change from 35.2% to 36.5%. This difference is not surprising, since the freshman class turns over completely each year and reflects admissions outcomes more quickly than the full undergraduate population, which still includes several earlier cohorts. In that sense, the 2024 first-year class may offer an early signal of how Rutgers’ student body could continue to evolve in the coming years.

At the same time, it is important not to over-interpret a single year of data or attribute changes directly to the Supreme Court decision. The early evidence at Rutgers does not show a decline in diversity, and the broader trend remains upward. However, the effects of policy changes often take time to appear, especially as older cohorts graduate and new admissions cycles occur under different rules. Continued observation of enrollment data will therefore be important for understanding how universities adapt to the new legal landscape.

Conclusion

Rutgers’ enrollment patterns show a clear rise in diversity over the past decade, with both the overall undergraduate population and the newest first-year class continuing that upward trend into 2024. The first year after the SFFA ruling does not show an immediate decline, and the freshman cohort in particular becomes more diverse rather than less. At the same time, a single year of post-decision data is only an early snapshot, and admissions changes often take several cycles to fully appear in enrollment patterns. Continued monitoring will therefore be important as institutions adapt to the new legal landscape and work to maintain broad access and representation.
